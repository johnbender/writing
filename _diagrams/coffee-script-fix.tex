\section{Grammar}

\begin{grammar}
<\lambda t> ::= \lit*{() ->} <t>
  \alt \lit*{->} <t>

<v> ::= \lit*{true}
  \alt \lit*{false}
  \alt <\lambda t>

<t> ::= <v>
  \alt <\lambda t>\lit*{()}
  \alt <\lambda t> <t>
\end{grammar}


\section{Grammar Examples}

\begin{equation}
\lit*{true}
\end{equation}

\begin{equation}
\lambda \lit*{false}  \ \lit*{()}
\end{equation}

\begin{equation}
\lambda \lit*{true}  \ \lit*{true}
\end{equation}

\begin{equation}
\lambda\lambda \lit*{true}  \lit*{()} \ \lambda \lit*{true}
\end{equation}

\section{Evaluation Rules/Relation}

\begin{equation}
\lambda t \lit*{()} \rightarrow  t
\end{equation}

\begin{equation}
\lambda t \  v  \rightarrow  t
\end{equation}

\begin{equation}
\trfrac{
   t_{2}  \rightarrow  t_{2}'
}{
  \lambda t_{1} \  t_{2}  \rightarrow \lambda t_{1} \  t_{2}'
}
\end{equation}
